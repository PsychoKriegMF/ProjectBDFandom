
// Создание таблиц

CREATE TABLE Person(
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    name VARCHAR
    );
    
CREATE TABLE Info(
    InfoID INTEGER PRIMARY KEY AUTOINCREMENT, 
    Description VARCHAR,
    Link VARCHAR,
    PersonID INTEGER,
    FOREIGN KEY (PersonID) REFERENCES Person(id) ON DELETE CASCADE
    );
    
CREATE TABLE Images(
    ImagesID INTEGER PRIMARY KEY AUTOINCREMENT,
    Image BLOB,
    PersonID INTEGER,
    FOREIGN KEY (PersonID) REFERENCES Person(id) ON DELETE CASCADE
    );
